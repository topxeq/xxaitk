# SSH 创建远程目录
# 替换 YOUR_HOST / YOUR_PASSWORD 为实际值后运行

# --- 创建目录 ---

# 明文模式:
aitk 'SSH_{"host":"YOUR_HOST","port":22,"user":"root","password":"YOUR_PASSWORD","action":"mkdir","remote_path":"/tmp/newdir/nested"}'

# Stdin模式:
echo '{"host":"YOUR_HOST","port":22,"user":"root","password":"YOUR_PASSWORD","action":"mkdir","remote_path":"/tmp/newdir/nested"}' | aitk SSH
