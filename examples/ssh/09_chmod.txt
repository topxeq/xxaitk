# SSH 修改文件权限
# 替换 YOUR_HOST / YOUR_PASSWORD 为实际值后运行

# --- 修改权限 ---

# 明文模式:
aitk 'SSH_{"host":"YOUR_HOST","port":22,"user":"root","password":"YOUR_PASSWORD","action":"chmod","remote_path":"/tmp/script.sh","mode":"0755"}'

# Stdin模式:
echo '{"host":"YOUR_HOST","port":22,"user":"root","password":"YOUR_PASSWORD","action":"chmod","remote_path":"/tmp/script.sh","mode":"0755"}' | aitk SSH
