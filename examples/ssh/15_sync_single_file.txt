# SSH 同步单个文件
# 替换 YOUR_HOST / YOUR_PASSWORD 为实际值后运行

# --- 推送单个文件 ---

# 明文模式:
aitk 'SSH_{"host":"YOUR_HOST","port":22,"user":"root","password":"YOUR_PASSWORD","action":"sync","direction":"push","local_path":"/tmp/config.yaml","remote_path":"/tmp/config.yaml"}'

# --- 拉取单个文件 ---

# 明文模式:
aitk 'SSH_{"host":"YOUR_HOST","port":22,"user":"root","password":"YOUR_PASSWORD","action":"sync","direction":"pull","local_path":"/tmp/remote_log.txt","remote_path":"/var/log/app.log"}'
