# SSH 移动/重命名文件
# 替换 YOUR_HOST / YOUR_PASSWORD 为实际值后运行

# --- 移动文件 ---

# 明文模式:
aitk 'SSH_{"host":"YOUR_HOST","port":22,"user":"root","password":"YOUR_PASSWORD","action":"move","source":"/tmp/old.txt","target":"/tmp/new.txt"}'

# Stdin模式:
echo '{"host":"YOUR_HOST","port":22,"user":"root","password":"YOUR_PASSWORD","action":"move","source":"/tmp/old.txt","target":"/tmp/new.txt"}' | aitk SSH
