# HTTP POST
#
# 三种输入模式:
#   1. Hex:    aitk HTTPPOST_<hex>
#   2. 明文:   aitk 'HTTPPOST_{"url":"...","body":"..."}'
#   3. Stdin:  echo '{"url":"...","body":"..."}' | aitk HTTPPOST

# --- POST JSON数据 ---

# 明文模式:
aitk 'HTTPPOST_{"url":"https://httpbin.org/post","body":"{\"name\":\"aitk\"}","content_type":"application/json"}'

# Hex模式:
# 编码前: {"url":"https://httpbin.org/post","body":"{\"name\": \"aitk\"}","content_type":"application/json"}
aitk HTTPPOST_7b2275726c223a2268747470733a2f2f6874747062696e2e6f72672f706f7374222c22626f6479223a227b5c226e616d655c223a205c226169746b5c227d222c22636f6e74656e745f74797065223a226170706c69636174696f6e2f6a736f6e227d

# Stdin模式:
echo '{"url":"https://httpbin.org/post","body":"hello","content_type":"text/plain"}' | aitk HTTPPOST

# --- POST 表单数据 ---

# 明文模式:
aitk 'HTTPPOST_{"url":"https://httpbin.org/post","body":"key=value","content_type":"application/x-www-form-urlencoded"}'

# --- 从文件读取body ---

# 明文模式:
aitk 'HTTPPOST_{"url":"https://httpbin.org/post","source":"file","source_path":"/tmp/data.json","content_type":"application/json"}'
